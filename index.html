        let audioCtx = null;

        function blasterSound() {

            if (!audioCtx) audioCtx = new (window.AudioContext || window.webkitAudioContext)();

            const o = audioCtx.createOscillator(), g = audioCtx.createGain();

            o.type = 'sawtooth';

            o.frequency.setValueAtTime(900, audioCtx.currentTime);

            o.frequency.exponentialRampToValueAtTime(120, audioCtx.currentTime + 0.18);

            g.gain.setValueAtTime(0.25, audioCtx.currentTime);

            g.gain.exponentialRampToValueAtTime(0.001, audioCtx.currentTime + 0.2);

            o.connect(g); g.connect(audioCtx.destination);

            o.start(); o.stop(audioCtx.currentTime + 0.2);

        }

        function hitSound() {

            if (!audioCtx) return;

            const o = audioCtx.createOscillator(), g = audioCtx.createGain();

            o.type = 'square';

            o.frequency.setValueAtTime(200, audioCtx.currentTime);

            o.frequency.exponentialRampToValueAtTime(50, audioCtx.currentTime + 0.25);

            g.gain.setValueAtTime(0.3, audioCtx.currentTime);

            g.gain.exponentialRampToValueAtTime(0.001, audioCtx.currentTime + 0.3);

            o.connect(g); g.connect(audioCtx.destination);

            o.start(); o.stop(audioCtx.currentTime + 0.3);

        }

        function overheatSound() {

            if (!audioCtx) return;

            o.frequency.exponentialRampToValueAtTime(80, audioCtx.currentTime + 0.5);

            o.connect(g); g.connect(audioCtx.destination);

            o.start(); o.stop(audioCtx.currentTime + 0.5);

        }

        BABYLON.SceneLoader.LoadAssetContainer('./', 'stormtrooper.glb', scene, function(loaded) {

            let max = new BABYLON.Vector3(-Infinity, -Infinity, -Infinity);

            container.meshes.forEach(m => {

                const b = m.getBoundingInfo().boundingBox;

            hasRunAnim = container.animationGroups.length > 0;

            spawnWave();

        });


        function spawnTrooper() {

            const inst = container.instantiateModelsToScene(name => name, false);

            const root = inst.rootNodes[0];

            root.scaling.scaleInPlace(trooperScale);


            const angle = Math.random() * Math.PI * 2;

            const dist = 20 + Math.random() * 15;

            root.position = new BABYLON.Vector3(

                camera.position.x + Math.sin(angle) * dist, 0,

                camera.position.z + Math.cos(angle) * dist

            );


            let anim = null;

            if (inst.animationGroups.length > 0) {

                anim = inst.animationGroups.find(a => /run|walk|move|sprint/i.test(a.name)) || inst.animationGroups[0];

                anim.start(true);

            }


            const childMeshes = root.getChildMeshes();

            childMeshes.forEach(m => { m.isPickable = true; });

            const trooper = { root, meshes: childMeshes, alive: true, anim, bobPhase: Math.random() * Math.PI * 2 };

            childMeshes.forEach(m => m.metadata = { trooper });

            troopers.push(trooper);

        }

        // ---------- KILL ----------

        function killTrooper(trooper, hitPoint) {

            trooper.alive = false;

            if (trooper.anim) trooper.anim.stop();

            hitSound();

            screenShake();

            sparkExplosion(hitPoint || trooper.root.position);


            score++;

            scoreEl.textContent = score;

            if (score % 5 === 0) { wave++; waveEl.textContent = wave; }


            const root = trooper.root;

            const tipAnim = new BABYLON.Animation('tip', 'rotation.x', 60,

                BABYLON.Animation.ANIMATIONTYPE_FLOAT, BABYLON.Animation.ANIMATIONLOOPMODE_CONSTANT);

            tipAnim.setKeys([{ frame: 0, value: root.rotation.x }, { frame: 30, value: root.rotation.x + Math.PI / 2 }]);

            root.animations = [tipAnim];

            scene.beginAnimation(root, 0, 30, false);


            setTimeout(() => {

                root.dispose();

                troopers = troopers.filter(t => t !== trooper);

                spawnTrooper();

                if (score % 5 === 0) spawnTrooper();

            }, 1000);

        }


        function impactFlash(position) {

            const flash = new BABYLON.PointLight('flash', position.clone(), scene);

            flash.diffuse = new BABYLON.Color3(1, 0.4, 0.15);

            flash.specular = new BABYLON.Color3(1, 0.5, 0.3);

            flash.intensity = 12;

            flash.range = 16;

            let life = 1;

            const obs = scene.onBeforeRenderObservable.add(() => {

                life -= engine.getDeltaTime() / 150;

                flash.intensity = 12 * Math.max(0, life);

                if (life <= 0) {

                    scene.onBeforeRenderObservable.remove(obs);

                    flash.dispose();

                }

            });

        }


        function sparkExplosion(position) {

            const ps = new BABYLON.ParticleSystem('sparks', 150, scene);

            ps.particleTexture = new BABYLON.Texture('https://assets.babylonjs.com/textures/flare.png', scene);

            ps.emitter = position.clone();

            ps.minEmitBox = new BABYLON.Vector3(-0.2, 0, -0.2);

            ps.maxEmitBox = new BABYLON.Vector3(0.2, 1.5, 0.2);

            ps.color1 = new BABYLON.Color4(1, 0.9, 0.3, 1);

            ps.color2 = new BABYLON.Color4(1, 0.4, 0.1, 1);

            ps.colorDead = new BABYLON.Color4(0.2, 0, 0, 0);

            ps.minSize = 0.08; ps.maxSize = 0.4;

            ps.minLifeTime = 0.2; ps.maxLifeTime = 0.6;

            ps.emitRate = 1200;

            ps.direction1 = new BABYLON.Vector3(-2, 2, -2);

            ps.direction2 = new BABYLON.Vector3(2, 4, 2);

            ps.gravity = new BABYLON.Vector3(0, -9.8, 0);

            ps.start();

            setTimeout(() => { ps.stop(); setTimeout(() => ps.dispose(), 800); }, 200);

        }


        let shakeT = 0;

        function screenShake() { shakeT = 1; }

        let bobT = 0;

        const lastPos = camera.position.clone();


        // ---------- GAME LOOP ----------

        scene.onBeforeRenderObservable.add(() => {

            const dt = engine.getDeltaTime() / 1000;


            // Heat cooldown

            if (heat > 0) {

                heat -= (overheated ? OVERHEAT_COOL_RATE : COOL_RATE) * dt;

                if (heat <= 0) {

                    heat = 0;

                    if (overheated) {

                        overheated = false;

                        overheatMsg.style.display = 'none';

                    }

                }

            }

            heatBar.style.width = heat + '%';


            // Bolts

            for (let i = bolts.length - 1; i >= 0; i--) {

                const b = bolts[i];

                const step = BOLT_SPEED * dt;

                const ray = new BABYLON.Ray(b.mesh.position, b.direction, step + 0.6);

                const hit = scene.pickWithRay(ray, m => m.isPickable && m.metadata && m.metadata.trooper);

                if (hit && hit.hit && hit.pickedMesh.metadata) {

                    const trooper = hit.pickedMesh.metadata.trooper;

                    if (trooper.alive) killTrooper(trooper, hit.pickedPoint);

                    impactFlash(b.mesh.position);

                    b.light.dispose(); b.mesh.dispose(); bolts.splice(i, 1); continue;

                }

                b.mesh.position.addInPlace(b.direction.scale(step));

                b.distanceLeft -= step;

                if (b.distanceLeft <= 0) { b.light.dispose(); b.mesh.dispose(); bolts.splice(i, 1); }

            }


            // ADS

            const targetFov = isADS ? ADS_FOV : BASE_FOV;

            camera.fov += (targetFov - camera.fov) * Math.min(1, dt * 12);

            const targetGun = isADS ? GUN_ADS : GUN_HIP;

            gun.position = BABYLON.Vector3.Lerp(gun.position, targetGun, Math.min(1, dt * 12));


            // Movement bob

            const horizMoved = Math.hypot(camera.position.x - lastPos.x, camera.position.z - lastPos.z);

            if (horizMoved > 0.001 && grounded) {

                bobT += dt * (sprinting ? 13 : 8);

                gun.position.y += Math.sin(bobT) * (sprinting ? 0.03 : 0.015);

                gun.position.x += Math.cos(bobT * 0.5) * 0.01;

                camera.position.y += Math.sin(bobT) * 0.015;

            }

            lastPos.set(camera.position.x, camera.position.y, camera.position.z);


            // Recoil

            if (recoilT > 0) {

                recoilT = Math.max(0, recoilT - dt * 8);

                gun.position.z -= recoilT * 0.15;

            }


            // Jump + crouch

            const targetH = crouching ? CROUCH_H : STAND_H;

            if (!grounded) {

                velY -= 14 * dt;

                camera.position.y += velY * dt;

                if (camera.position.y <= targetH) {

                    camera.position.y = targetH;

                    velY = 0;

                    grounded = true;

                }

            } else {

                camera.position.y += (targetH - camera.position.y) * Math.min(1, dt * 10);

            }

            camera.speed = crouching ? 0.15 : (isADS ? 0.18 : (sprinting ? 0.6 : 0.35));


            // Shake

            if (shakeT > 0) {

                shakeT = Math.max(0, shakeT - dt * 5);

                camera.rotation.z = (Math.random() - 0.5) * 0.02 * shakeT;

            } else {

                camera.rotation.z = 0;

            }


            if (gameOver) return;


            // Troopers advance

            const speed = (0.8 + wave * 0.15) * dt;

            troopers.forEach(t => {

                if (!t.alive) return;

                const toPlayer = camera.position.subtract(t.root.position);

                toPlayer.y = 0;

                const distance = toPlayer.length();

                if (distance < 1.5) { endGame(); return; }

                t.root.position.addInPlace(toPlayer.normalize().scale(speed));

                t.root.lookAt(new BABYLON.Vector3(camera.position.x, 0, camera.position.z));

                t.root.rotate(BABYLON.Axis.Y, -Math.PI / 2, BABYLON.Space.LOCAL);


                if (!hasRunAnim) {

                    t.bobPhase += dt * 10;

                    t.root.position.y = Math.abs(Math.sin(t.bobPhase)) * 0.12;

                    t.root.rotation.x = 0.15;

                }

            });

        });


        function endGame() {

            gameOver = true;

            document.exitPointerLock();

            document.getElementById('final-score').textContent = score;

            document.getElementById('gameover').style.display = 'flex';

        }


        engine.runRenderLoop(() => scene.render());

        window.addEventListener('resize', () => engine.resize());

    </script>
